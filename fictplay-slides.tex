from random import uniform, choice
import numpy as np

def fictplay(t):
    x_0_t = [uniform(0, 1)]
    x_1_t = [uniform(0, 1)]
    gain_0 = np.array([[1, -1], [-1, 1]])   #player0
    gain_1 = np.array([[-1, 1], [1, -1]])   #player1
